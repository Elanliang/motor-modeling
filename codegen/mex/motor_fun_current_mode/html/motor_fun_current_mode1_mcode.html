1function [ t, v_mag ] = motor_fun_current_mode(i_mag, i_phase, thetadot) %#codegen
  2
  3%%%%%% Specify Motor Parameters %%%%%%
  4%%%%%% Edit these parameters %%%%%%
  5%%% Flux Linkage, Wb %%%
  6k1 = 0.06; %%harmonic coefficients
  7k3 = .00;
  8k5 = -.00;
  9k7 = 0;
 10k9 = 0.000; 
 11%%% pm flux linked by rotor at angle theta_r to phase at angle theta_p %%%
 12wb_r = @(theta_r, theta_p) k1*cos(theta_p - theta_r) + k3*cos(3*(-theta_r) +theta_p) + k5*cos(5*(-theta_r)+theta_p) + k9*cos(9*(-theta_r) + theta_p);
 13
 14%%% Phase Resistances %%%
 15r_a = .1;
 16r_b = .1;
 17r_c = .1;
 18
 19%%% Termination Type: wye, delta, ind %%%
 20termination = 'wye';
 21
 22%%% Pole Pairs %%%
 23npp = 3;
 24
 25%%% Inductances %%%
 26l_d = .0008;     %D-Axis Inductance
 27l_q = .002;      %Q-Axis Inductance
 28l_m =  .00;    %Phase Mutual Inductance, assuming a constant for now
 29
 30
 31%%%%% Automatic Setup %%%%%
 32%%%%% Don't change unless you know what you're doing %%%%%
 33%%% Resistance Matrix %%%
 34R = [r_a 0 0; 
 35    0 r_b 0; 
 36    0 0 r_c];
 37
 38%%% Phase Self Inductance %%%
 39l_p = @(theta_r, theta_p) .5*(l_d - l_q)*(cos(2*(theta_p - theta_r)))+(l_d + l_q)/2;
 40
 41%%% Inductance Matrix %%% 
 42L = @(theta_r) [l_p(theta_r, 0), l_m, l_m; l_m, l_p(theta_r, 2*pi/3), l_m; l_m, l_m, l_p(theta_r, -2*pi/3)];
 43
 44%%% Flux Linkage Matrix %%%
 45Wb = @(theta_r, i) [L(theta_r)]*i + [wb_r(theta_r, 0); wb_r(theta_r, 2*pi/3); wb_r(theta_r, -2*pi/3)];
 46
 47
 48%%% Transforms %%%
 49%%% Power-invariant form %%%
 50%%% Not your canonical transform, but it fits my assumptions %%%
 51abc = @(theta) sqrt(2/3)*[cos(-theta), sin(-theta), 1/sqrt(2);
 52    cos((2*pi/3)-theta), sin((2*pi/3)-theta), 1/sqrt(2);
 53    cos((-2*pi/3)-theta), sin((-2*pi/3)-theta), 1/(sqrt(2))];
 54
 55dq0 = @(theta) abc(theta)';%inv(abc(theta));
 56
 57%%% Inverter Properties %%%
 58f_switch = 10000;    %%Loop frequency
 59v_bus = 160;         %%Bus voltage
 60
 61%%% Current Controller %%%
 62
 63i_q = i_mag*sin(i_phase);
 64i_d = i_mag*cos(i_phase);
 65
 66i_dq0 = [i_d; i_q; 0];
 67
 68%%% Initialize Dynamics Variables %%%
 69
 70v = [0; 0; 0];
 71v_bemf = v;
 72theta = 0;
 73i = abc(theta)*i_dq0;
 74i_old = [0; 0; 0];
 75i_dot = [0; 0; 0];
 76thetadotdot = 0;
 77phase_shift = 0;
 78
 79tfinal = .0002;
 80dt = 1/(f_switch);     %%Simulation time step
 81%dt = 1e-5;
 82t = 0:dt:tfinal;
 83wb_abc_rotor_old = [wb_r(theta, 0); wb_r(theta, 2*pi/3); wb_r(theta, -2*pi/3)];
 84
 85timer_step = (dt/(1/(2*f_switch)));
 86timer_dir = 1;
 87
 88wb_old = Wb(theta, i);
 89l_old = L(theta);
 90
 91thetadot_vec = zeros(length(t), 1);
 92v_vec = zeros(length(t), 3);
 93v_dq0_vec = zeros(length(t), 2);
 94i_vec = zeros(length(t), 3);
 95torque_vec = zeros(length(t), 1);
 96power_mech_vec = zeros(length(t), 1);
 97v_uvw_vec = zeros(length(t), 3);
 98
 99
100%tic
101for j=1:length(t)
102    time = t(j);
103 
104    %%% Sample Current %%%
105    
106    if (strcmp(termination, 'delta'))
107        i_sample = [i(1)-i(3); i(2) - i(1); i(3) - i(2)];
108    elseif (strcmp(termination, 'wye'))
109        i_sample = i;
110    elseif (strcmp(termination, 'ind'))
111        i_sample = i;
112    end
113    
114    %%% Calculate Transform Matrix %%%
115    %%dq0_transform = dq0(theta);
116    abc_transform = abc(theta);
117    dq0_transform = inv(abc_transform);
118    
119    i = abc_transform*i_dq0;
120    i_dot = (i - i_old)*(1/dt);
121    i_old = i;
122
123
124   
125
126    %%% Rotor Flux linked to each phase, and derivative %%
127    wb_abc_rotor = [wb_r(theta, 0); wb_r(theta, 2*pi/3); wb_r(theta, -2*pi/3)];
128    wb_abc_rotor_dot = (wb_abc_rotor - wb_abc_rotor_old)*(1/dt);
129    
130    %%% Phase Inductance and derivative %%%
131    l = L(theta);
132    l_dot = (l-l_old)*(1/dt);
133    
134    %%% Back-EMF %%%
135    v_bemf = (l_dot*i + wb_abc_rotor_dot);
136    
137    
138    %%% Phase Voltages %%%
139     value = [l, [1;1;1]; [1 1 1 0]]*[i_dot; 0] + [v_bemf;0] + [R, [0; 0; 0]; [0 0 0 0]]*[i; 0];
140     v_uvw = value(1:3);
141     v_dq0 = dq0_transform*v_uvw;
142     v_dq0 = v_dq0(1:2);
143
144    %%% Mechanical Power %%%
145    p_mech_abc =  v_bemf.*i;
146    p_mech = sum(p_mech_abc);
147    
148    %%% Phase Torques %%%
149    torque_abc = p_mech_abc*(1/thetadot);
150    
151    %%% Total Torque %%%
152    torque = sum(torque_abc);
153
154    thetadot = thetadot + thetadotdot*dt;
155    thetadot_mech = thetadot;
156    
157    theta = theta + thetadot*dt;
158
159   
160    wb_abc_rotor_old = wb_abc_rotor;
161    l_old = l;
162    
163    %%% Save Data %%%
164    thetadot_vec(j) = thetadot;
165    i_vec(j,:) = i';
166    v_vec(j,:) = v'; 
167    v_uvw_vec(j,:) = v_uvw';
168    v_dq0_vec(j, :) = v_dq0';
169    torque_vec(j) = torque;
170    power_mech_vec(j) = p_mech;
171    %current_mag_vec(j) = sample_mag;
172end
173%toc
174
175%figure;plot(thetadot_vec, i_vec);
176%figure;plot(t, v_bemf_vec);
177%figure;plot(thetadot_vec, i_dq_vec); title('I D/Q');
178%figure;plot(t, torque_vec); title ('Torque');
179%hold all; plot(t, torque_pm_vec); plot(t, torque_rel_vec);
180%figure;plot(t, thetadot_mech_vec); title('Theta dot');
181%figure;plot(thetadot_mech_vec, torque_vec); title('Torque vs Speed');
182%figure;plot(t, v_uvw_vec); title('UVW Voltages');
183%figure;plot(t, v_dq0_vec); title('dq0 voltages');
184%figure;plot(thetadot_mech_vec, power_mech_vec); title('Power vs Speed');
185%figure;plotyy(t, phase_shift_vec, t, current_mag_vec);title('Current Phase/Mag');
186%figure;plot(t, torque_abc_vec);
187
188t = torque_vec(end);
189v_mag = norm(v_dq0);
190
191%toc
192
193end
